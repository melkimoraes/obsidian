SELECT CAB.NUNOTA,
TO_CHAR(CAB.DTFATUR,'DD/MM/YY HH24:MI:SS'),
CABR.NUNOTA,
--CABM.CODPARC,
ITER.CODPROD, 
ITER.QTDNEG,
SUM(ITEM.QTDNEG) 
FROM AD_PROCREM REM
INNER JOIN TGFCAB CAB ON CAB.NUNOTA=REM.NUNOTA
INNER JOIN TGFCAB CABR ON CABR.NUNOTA = REM.NUNOTAREM
INNER JOIN TGFITE ITER ON ITER.NUNOTA = CABR.NUNOTA
INNER JOIN AD_PROCREMMOV REMMV ON REMMV.NUNOTA = REM.NUNOTA
INNER JOIN TGFCAB CABM ON CABM.NUNOTA = REMMV.NUNOTAMOV
INNER JOIN TGFITE ITEM ON ITEM.NUNOTA = REMMV.NUNOTAMOV
WHERE CAB.CODEMP = 2
AND TRUNC(CAB.DTFATUR) >= TRUNC(TO_DATE('30/05/25','DD/MM/YY')) --and cab.nunota not in (2403369)
AND ITER.CODPROD = 423
GROUP BY CAB.NUNOTA,CAB.DTFATUR,CABR.NUNOTA,ITER.CODPROD, ITER.QTDNEG
ORDER BY 1 DESC
